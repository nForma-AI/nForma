    **activity-set `<json>`:**
    - Parse the JSON string argument
    - Ensure `updated` field is always set to `new Date().toISOString()` (overrides any caller-provided value for consistency)
    - Write to `.planning/current-activity.json` via `fs.writeFileSync` (atomic on POSIX filesystems)
    - Output: `{ "written": true, "path": ".planning/current-activity.json" }`
    - Error if JSON is malformed: exit 1 with stderr message

    **activity-clear:**
    - Remove `.planning/current-activity.json` if it exists
    - No error if file is absent (idempotent)
    - Output: `{ "cleared": true }`

    **activity-get:**
    - Read `.planning/current-activity.json`
    - If file exists: output its JSON content as parsed object
    - If file is missing: output `{}` (empty object, no error)

    Test cases:
    - TC1: activity-set with full schema → file written with all fields + `updated` timestamp
    - TC2: activity-set with minimal fields (activity only) → file written with just `activity` + `updated`
    - TC3: activity-clear removes file → activity-get then returns `{}`
    - TC4: activity-clear on non-existent file → returns `{ "cleared": true }` without error
  </behavior>
  <implementation>
    Add three handler functions in gsd-tools.cjs:

    ```javascript
    function cmdActivitySet(cwd, jsonStr, raw) {
      let data;
      try {
        data = JSON.parse(jsonStr);
      } catch (e) {
        error(`activity-set: invalid JSON — ${e.message}`);
      }
      data.updated = new Date().toISOString();
      const filePath = path.join(cwd, '.planning', 'current-activity.json');
      fs.writeFileSync(filePath, JSON.stringify(data, null, 2), 'utf-8');
      output({ written: true, path: '.planning/current-activity.json' }, raw);
    }

    function cmdActivityClear(cwd, raw) {
      const filePath = path.join(cwd, '.planning', 'current-activity.json');
      try { fs.unlinkSync(filePath); } catch (e) { /* idempotent */ }
      output({ cleared: true }, raw);
    }

    function cmdActivityGet(cwd, raw) {
      const filePath = path.join(cwd, '.planning', 'current-activity.json');
      try {
        const content = fs.readFileSync(filePath, 'utf-8');
        output(JSON.parse(content), raw);
      } catch (e) {
        output({}, raw);
      }
    }
    ```

    Wire into the command dispatch switch at the bottom of main():
    ```javascript
    case 'activity-set': {
      cmdActivitySet(cwd, args[1], raw);
      break;
    }
    case 'activity-clear': {
      cmdActivityClear(cwd, raw);
      break;
    }
    case 'activity-get': {
      cmdActivityGet(cwd, raw);
      break;
    }
    ```

    Also update the header comment block at the top of the file to document the new commands.

    For tests: add a `describe('activity commands', ...)` block to gsd-tools.test.cjs following the existing test patterns (createTempProject, cleanup, runGsdTools).
