**Code evidence:**

```javascript
function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));

// line 136-157 (isRetryable classification)
function isRetryable(error) {
  const msg = error?.message || String(error);
  if (/usage:|unknown flag|unknown option|invalid flag|unrecognized/i.test(msg)) {
    return false; // CLI_SYNTAX errors not retryable
  }
  if (/spawn error/i.test(msg)) {
    return false; // spawn errors not retryable
  }
  if (/TIMEOUT/i.test(msg)) {
    return true; // TIMEOUT retryable
  }
  if (/ECONNREFUSED|ENOTFOUND|ECONNRESET|ETIMEDOUT/i.test(msg)) {
    return true; // network errors retryable
  }
  return true; // fail-open: unknown errors retryable
}

// line 159-183 (retryWithBackoff function)
async function retryWithBackoff(fn, slotName, maxRetries = 2, delays = [1000, 3000]) {
  const MAX_RETRIES = maxRetries;
  let retryAttempts = 0;
  for (let attempt = 0; attempt <= MAX_RETRIES; attempt++) {
    try {
      const result = await fn();
      return { result, retryCount: retryAttempts };
    } catch (err) {
      const isLastAttempt = attempt >= MAX_RETRIES;
      const isNonRetryable = !isRetryable(err);
      if (isNonRetryable || isLastAttempt) {
        throw err; // fail immediately if non-retryable or last attempt
      }
      const delayMs = delays[attempt] ?? 3000;
      retryAttempts++;
      process.stderr.write(`[call-quorum-slot] retry ${attempt + 1}/${MAX_RETRIES} for slot ${slotName} after ${delayMs}ms\n`);
      await sleep(delayMs);
    }
  }
}
```

- HTTP (line 484): `const retryResult = await retryWithBackoff(() => runHttp(...), slot);`

### Truth 3: Provider-Aware Dispatch Skip

**Code evidence from qgsd-prompt.js:**

```javascript
// line 260-303 (getDownProviderSlots function)
function getDownProviderSlots() {
  try {
    const cachePath = path.join(os.homedir(), '.claude', 'qgsd-provider-cache.json');
    if (!fs.existsSync(cachePath)) return [];
    const cache = JSON.parse(fs.readFileSync(cachePath, 'utf8'));
    if (!cache || !cache.entries) return [];

    const providers = findProviders();
    if (!providers) return [];

    // Extract hostnames from DOWN cache entries
    const downHostnames = [];
    const now = Date.now();
    for (const [baseUrl, entry] of Object.entries(cache.entries)) {
      const ttl = entry.healthy ? 180000 : 300000;
      if (now - entry.cachedAt >= ttl) continue; // expired — ignore
      if (entry.healthy) continue; // UP — not skipping

      try {
        const hostname = new URL(baseUrl).hostname; // e.g., "api.akashml.com"
        let normalized = hostname.replace(/^api\./, ''); // remove "api." prefix
        normalized = normalized.replace(/\.(com|ai|xyz|io)$/, ''); // remove TLD
        downHostnames.push(normalized);
      } catch (_) { /* malformed URL — skip */ }
    }

    if (downHostnames.length === 0) return [];

    // Find slots backed by down providers
    const skipSlots = [];
    for (const p of providers) {
      if (!p.provider) continue;
      for (const dh of downHostnames) {
        if (p.provider.includes(dh) || dh.includes(p.provider)) {
          skipSlots.push(p.name);
          break;
        }
      }
    }
    return skipSlots;
  } catch (_) {
    return []; // fail-open: any error returns empty skip list
  }
}

// line 437-439 (Dispatch integration)
const providerSkips = getDownProviderSlots();
const allSkipSlots = [...new Set([...recentTimeouts, ...providerSkips])];

}
```

**Logic flow:**
1. Read provider health cache at ~/.claude/qgsd-provider-cache.json
2. For each DOWN provider (healthy=false, within TTL):
   - Extract hostname from baseUrl (e.g., api.akashml.com → akashml)
   - Find all slots in providers.json with matching provider field
   - Add to skip list
3. Remove all skipped slots from cappedSlots before building dispatch
4. Add SKIP note to quorum instructions
5. Fail-open: any error returns empty skip list (dispatch proceeds normally)
