    'nf-statusline',
  ]),

Note: `standard` and `strict` have the SAME hook set. The behavioral difference (strict expands quorum to all /nf: commands) is handled inside nf-prompt.js and nf-stop.js, not by the profile map.

Note: `nf-check-update` is not in any profile map -- it is a SessionStart hook that runs independently of the profile system.

2. **shouldRunHook() function** (place after HOOK_PROFILE_MAP):
```javascript
function shouldRunHook(hookBasename, profile) {
  const validProfile = HOOK_PROFILE_MAP[profile] ? profile : 'standard';
  return HOOK_PROFILE_MAP[validProfile].has(hookBasename);
}
```

3. **DEFAULT_CONFIG addition** -- add `hook_profile: 'standard'` to DEFAULT_CONFIG object (after `task_envelope_enabled`).

4. **validateConfig() addition** -- add hook_profile validation after the existing `task_envelope_enabled` validation block:
```javascript
const VALID_PROFILES = ['minimal', 'standard', 'strict'];
if (config.hook_profile !== undefined) {
  if (typeof config.hook_profile !== 'string' || !VALID_PROFILES.includes(config.hook_profile)) {
    process.stderr.write('[nf] WARNING: nf.json: hook_profile must be "minimal", "standard", or "strict"; defaulting to "standard"\n');
    config.hook_profile = 'standard';
  }
}
```

5. **module.exports** -- add `shouldRunHook` and `HOOK_PROFILE_MAP` to the existing exports:
```javascript
module.exports = { loadConfig, DEFAULT_CONFIG, SLOT_TOOL_SUFFIX, slotToToolCall, shouldRunHook, HOOK_PROFILE_MAP };
```

CRITICAL: Do NOT introduce any module-level config caching. loadConfig() must continue reading nf.json from disk on every call. This is what satisfies PROF-04.
