```javascript
// ── Cold-start state detection (CALIB-02, CALIB-03) ─────────────────────
function computeColdStartState(pol, sbPath, crPath) {
  let ciRunCount = 0;
  let quorumRoundCount = 0;
  let firstRunTimestamp = null;

  // Count CI runs: number of lines in check-results.ndjson
  if (fs.existsSync(crPath)) {
    const lines = fs.readFileSync(crPath, 'utf8')
      .trim().split('\n').filter(l => l.length > 0);
    ciRunCount = lines.length;
  }

  // Read quorum rounds from scoreboard
  if (fs.existsSync(sbPath)) {
    try {
      const sb = JSON.parse(fs.readFileSync(sbPath, 'utf8'));
      const rounds = Array.isArray(sb.rounds) ? sb.rounds : [];
      quorumRoundCount = rounds.length;
      if (rounds.length > 0) {
        const firstRound = rounds[0];
        // Support timestamp (ISO) or date (MM-DD) field
        const raw = firstRound.timestamp || firstRound.date;
        if (raw) {
          const parsed = Date.parse(raw);
          if (!isNaN(parsed)) firstRunTimestamp = parsed;
        }
      }
    } catch (e) {
      process.stderr.write('[run-prism] Warning: failed to parse scoreboard for cold-start: ' + e.message + '\n');
    }
  }

  // Compute days since first run (0 if no history)
  const daysSinceFirst = firstRunTimestamp
    ? (Date.now() - firstRunTimestamp) / (1000 * 60 * 60 * 24)
    : 0;

  // Cold-start is true if ANY threshold is unmet
  const allThresholdsMet =
    ciRunCount       >= pol.cold_start.min_ci_runs &&
    quorumRoundCount >= pol.cold_start.min_quorum_rounds &&
    daysSinceFirst   >= pol.cold_start.min_days;
  const inColdStart = !allThresholdsMet;

  return { inColdStart, ciRunCount, quorumRoundCount, daysSinceFirst, firstRunTimestamp };
}
```

```javascript
function makePolicyYaml({ minCiRuns = 5, minQuorumRounds = 10, minDays = 1, mode = 'warn' } = {}) {
  return [
    'min_ci_runs: ' + minCiRuns,
    'min_quorum_rounds: ' + minQuorumRounds,
    'min_days: ' + minDays,
    'mode: "' + mode + '"',
    'tp_rate: 0.85',
    'unavail: 0.15',
  ].join('\n');
}

function makeNdjsonLines(count) {
  return Array.from({ length: count }, (_, i) =>
    JSON.stringify({ tool: 'run-prism', formalism: 'prism', result: 'pass', timestamp: new Date().toISOString(), metadata: {} })
  ).join('\n') + '\n';
}
```

Helper to set up a test directory with policy.yaml, optional scoreboard, optional check-results.ndjson, and optionally the formal/prism/quorum.pm model:
```javascript
function setupTestDir(opts = {}) {
  const tmp = fs.mkdtempSync(path.join(os.tmpdir(), 'run-prism-calib-'));
  // policy.yaml required in formal/
  fs.mkdirSync(path.join(tmp, 'formal', 'prism'), { recursive: true });
  fs.mkdirSync(path.join(tmp, '.planning'), { recursive: true });
  fs.writeFileSync(path.join(tmp, 'formal', 'policy.yaml'), makePolicyYaml(opts.policy || {}));

  if (opts.scoreboardRounds !== undefined) {
    const rounds = Array.from({ length: opts.scoreboardRounds }, (_, i) => ({
      date: '2026-02-27', timestamp: new Date(Date.now() - i * 1000).toISOString(),
      task: 'r' + i, round: 1, votes: { claude: 'TP', gemini: 'TP' }
    }));
    fs.writeFileSync(
      path.join(tmp, '.planning', 'quorum-scoreboard.json'),
      JSON.stringify({ models: {}, rounds })
    );
  }

  if (opts.ciRunCount !== undefined) {
    fs.writeFileSync(
      path.join(tmp, 'formal', 'check-results.ndjson'),
      makeNdjsonLines(opts.ciRunCount)
    );
  }

  return tmp;
}
```

```javascript
test('cold-start: all thresholds unmet → stderr notes cold-start active', () => {
  const tmp = setupTestDir({ policy: { minCiRuns: 5, minQuorumRounds: 10, minDays: 1 } });

test('observation_window: metadata included in NDJSON entry', () => {
  const ndjsonPath = path.join(os.tmpdir(), 'test-obs-' + Date.now() + '.ndjson');

    spawnSync(process.execPath, [RUN_PRISM], {

    if (fs.existsSync(ndjsonPath)) {

test('observation_window: timestamps are ISO 8601 format', () => {
  const ndjsonPath = path.join(os.tmpdir(), 'test-iso-' + Date.now() + '.ndjson');
  const tmp = setupTestDir({
    policy: { minCiRuns: 1, minQuorumRounds: 1, minDays: 0 },
    scoreboardRounds: 5, ciRunCount: 5
  });
  try {
    spawnSync(process.execPath, [RUN_PRISM], {

    if (fs.existsSync(ndjsonPath)) {
      const line = fs.readFileSync(ndjsonPath, 'utf8').trim().split('\n').find(l => l);

          assert.match(window_end,   ISO_RE, 'window_end must be ISO 8601');

      }
    }
  } finally {
    fs.rmSync(tmp, { recursive: true, force: true });
    if (fs.existsSync(ndjsonPath)) fs.unlinkSync(ndjsonPath);
  }
```
