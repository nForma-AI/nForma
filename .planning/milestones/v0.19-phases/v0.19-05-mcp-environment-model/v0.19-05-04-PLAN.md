    // formal/prism/mcp-availability.pm
    // MCPENV-04: Per-slot MCP availability model calibrated from scoreboard.
    // Parameters (const) are injected at runtime by run-prism.cjs from quorum-scoreboard.json.
    // Fall-back priors used when scoreboard has insufficient data.
    //
    // Usage: prism formal/prism/mcp-availability.pm formal/prism/mcp-availability.props
    //   -const codex_1_avail=0.95 -const gemini_1_avail=0.92 ...

    dtmc

    // Per-slot availability rates (injected from scoreboard at runtime)
    const double codex_1_avail    = 0.85;  // prior; overridden by run-prism.cjs
    const double gemini_1_avail   = 0.85;
    const double opencode_1_avail = 0.85;
    const double copilot_1_avail  = 0.85;

    module mcp_slots
      codex_avail    : [0..1] init 1;  // 1 = available, 0 = unavailable
      gemini_avail   : [0..1] init 1;
      opencode_avail : [0..1] init 1;
      copilot_avail  : [0..1] init 1;

      // Each slot independently transitions at each step
      [] (codex_avail=1)    -> codex_1_avail    : (codex_avail'=1)
                             + (1-codex_1_avail) : (codex_avail'=0);
      [] (codex_avail=0)    -> codex_1_avail    : (codex_avail'=1)
                             + (1-codex_1_avail) : (codex_avail'=0);
      [] (gemini_avail=1)   -> gemini_1_avail   : (gemini_avail'=1)
                             + (1-gemini_1_avail): (gemini_avail'=0);
      [] (gemini_avail=0)   -> gemini_1_avail   : (gemini_avail'=1)
                             + (1-gemini_1_avail): (gemini_avail'=0);
      [] (opencode_avail=1) -> opencode_1_avail : (opencode_avail'=1)
                             + (1-opencode_1_avail): (opencode_avail'=0);
      [] (opencode_avail=0) -> opencode_1_avail : (opencode_avail'=1)
                             + (1-opencode_1_avail): (opencode_avail'=0);
      [] (copilot_avail=1)  -> copilot_1_avail  : (copilot_avail'=1)

      [] (copilot_avail=0)  -> copilot_1_avail  : (copilot_avail'=1)

    endmodule

    // At least one slot is available (minimum quorum for any response)
    label "min_quorum_available" =

    ```javascript
    // Reads quorum-scoreboard.json and computes per-slot availability rates.
    // Returns { 'slot-name': availabilityRate, ... } or null if no data.
    // Rate = 1.0 - (unavail_count / total_count) per slot, excluding 'claude' (self).
    // MCPENV-04
    function readMCPAvailabilityRates(sbPath) {
      const p = sbPath || path.join(process.cwd(), '.planning', 'quorum-scoreboard.json');
      try {
        const raw = fs.readFileSync(p, 'utf8');
        const sb = JSON.parse(raw);
        const rounds = Array.isArray(sb.rounds) ? sb.rounds : [];
        if (rounds.length === 0) return null;

        const slotStats = {};
        for (const round of rounds) {
          const votes = round.votes || {};
          for (const [slot, code] of Object.entries(votes)) {
            if (slot === 'claude') continue;  // exclude self
            if (!slotStats[slot]) slotStats[slot] = { total: 0, unavail: 0 };
            slotStats[slot].total++;
            if (code === 'UNAVAIL') slotStats[slot].unavail++;
          }
        }

        const rates = {};
        for (const [slot, stats] of Object.entries(slotStats)) {
          if (stats.total === 0) continue;
          rates[slot] = Math.round((1.0 - stats.unavail / stats.total) * 1e6) / 1e6;
        }
        return Object.keys(rates).length > 0 ? rates : null;
      } catch (_) {
        return null;  // missing or malformed scoreboard — caller uses priors
      }
    }
    ```
