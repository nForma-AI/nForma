**1. NESTED_TO_FLAT_MAP constant:**

Define a mapping of known nested key paths to their flat equivalents. Derive this from DEFAULT_CONFIG -- keys that exist as top-level flat keys but could be expressed nested. The config uses shallow merge, so nested objects get replaced entirely. The adapter must convert between formats.

```javascript
// Bidirectional map: nested path -> flat key.
// Covers keys that users might write in nested form but loadConfig expects flat.
// Example: { model_tier: { planner: 'opus' } } -> { model_tier_planner: 'opus' }
const NESTED_TO_FLAT_MAP = {
  'model_tier.planner': 'model_tier_planner',
  'model_tier.worker': 'model_tier_worker',
  'smart_compact.threshold_pct': 'smart_compact_threshold_pct',
};

// Reverse map: flat key -> { parent, child }
const FLAT_TO_NESTED_MAP = {};
for (const [nestedPath, flatKey] of Object.entries(NESTED_TO_FLAT_MAP)) {
  const [parent, child] = nestedPath.split('.');
  FLAT_TO_NESTED_MAP[flatKey] = { parent, child };
}
```

**2. flattenNestedKeys(config):**
```javascript
function flattenNestedKeys(config) {
  if (typeof config !== 'object' || config === null) return {};
  const result = { ...config };

  // For each known nested path, check if user wrote the nested form
  for (const [nestedPath, flatKey] of Object.entries(NESTED_TO_FLAT_MAP)) {
    const [parent, child] = nestedPath.split('.');
    if (result[parent] && typeof result[parent] === 'object' && child in result[parent]) {
      // Only flatten if the flat key is NOT already set (nested form is a fallback)
      if (!(flatKey in result)) {
        result[flatKey] = result[parent][child];
      }
      // Remove the child from the nested object (leave other children intact)
      const nested = { ...result[parent] };
      delete nested[child];
      if (Object.keys(nested).length === 0) {
        // If parent object is now empty AND parent is not a known DEFAULT_CONFIG key with sub-keys, remove it
        // But if parent like 'quorum' has other legitimate sub-keys (minSize, preferSub), keep it
      } else {
        result[parent] = nested;
      }
    }
  }
  return result;
}
```

Note: The `model_tier` parent key does NOT exist in DEFAULT_CONFIG as a nested object -- `model_tier_planner` and `model_tier_worker` are flat keys. So if a user writes `{ model_tier: { planner: 'opus', worker: 'haiku' } }`, both children get flattened and the empty `model_tier` parent is removed. For `smart_compact`, the parent IS a known nested object in DEFAULT_CONFIG (with `enabled` and `context_warn_pct` sub-keys), so only the `threshold_pct` child is extracted and the parent is preserved with remaining children.

**3. nestFlatKeys(config):**
```javascript
function nestFlatKeys(config) {
  if (typeof config !== 'object' || config === null) return {};
  const result = { ...config };

  for (const [flatKey, { parent, child }] of Object.entries(FLAT_TO_NESTED_MAP)) {
    if (flatKey in result) {
      if (!result[parent] || typeof result[parent] !== 'object') {
        result[parent] = {};
      }
      result[parent] = { ...result[parent], [child]: result[flatKey] };
      delete result[flatKey];
    }
  }
  return result;
}
```

**4. normalizeConfigValue(key, value):**
```javascript
function normalizeConfigValue(key, value) {
  // Boolean string normalization: "true"/"false" -> true/false
  if (typeof value === 'string') {
    const lower = value.toLowerCase();
    if (lower === 'true') return true;
    if (lower === 'false') return false;
  }
  // Profile and tier case normalization
  const CASE_NORMALIZED_KEYS = ['hook_profile', 'model_tier_planner', 'model_tier_worker', 'fail_mode'];
  if (CASE_NORMALIZED_KEYS.includes(key) && typeof value === 'string') {
    return value.toLowerCase();
  }
  // Pass through all other values unchanged
  return value;
}
```

**5. normalizeConfig(config):**
```javascript
function normalizeConfig(config) {
  if (typeof config !== 'object' || config === null) return {};
  // Step 1: Flatten any nested keys to their flat equivalents
  const flattened = flattenNestedKeys(config);
  // Step 2: Normalize individual values (booleans, case)
  const result = {};
  for (const [key, value] of Object.entries(flattened)) {
    result[key] = normalizeConfigValue(key, value);
  }
  return result;
}
```

**6. writeConfig(filePath, config):**

```javascript
function writeConfig(filePath, config) {
  const normalized = normalizeConfig(config);
  fs.writeFileSync(filePath, JSON.stringify(normalized, null, 2) + '\n', 'utf8');
}
```
