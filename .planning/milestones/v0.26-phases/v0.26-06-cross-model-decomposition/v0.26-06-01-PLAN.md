Add four new pure functions to `bin/analyze-state-space.cjs`, placed between the existing `resolveCardinality()` function and the `// -- Model Analysis` section comment (around line 493).

**1. `extractRequirementPrefixes(requirements)`**
Takes an array of requirement ID strings (e.g., `["DETECT-01", "DETECT-02", "CRED-03"]`).
Returns a Set of prefix strings (e.g., `Set {"DETECT", "CRED"}`).
Algorithm: for each requirement ID, split on `-`, take all parts except the last (which is the numeric suffix), rejoin with `-`. Examples:
- `"DETECT-01"` -> `"DETECT"`
- `"CRED-07"` -> `"CRED"`
- `"PRM-AM-01"` -> `"PRM-AM"`

If `requirements` is undefined, null, or empty, return an empty Set.

**2. `findCandidatePairs(registry, modelAnalyses)`**
Takes the parsed `model-registry.json` and the `models` object from analysis (same shape as `report.models`).
Returns an array of pair objects: `{ model_a, model_b, shared_source_files, shared_requirement_prefixes, shared_requirements }`.

Algorithm:
- Get the list of TLA+ model keys from `modelAnalyses` (these are the analyzed models).
- For each key, look up the corresponding entry in `registry.models` to get `requirements` and `source_files` arrays.
- For each unique unordered pair `(a, b)` where `a < b` alphabetically:
  - `sharedSourceFiles`: intersection of `registry.models[a].source_files || []` and `registry.models[b].source_files || []`
  - `prefixesA`: `extractRequirementPrefixes(registry.models[a].requirements)`
  - `prefixesB`: `extractRequirementPrefixes(registry.models[b].requirements)`
  - `sharedPrefixes`: intersection of prefixesA and prefixesB (as array)
  - `sharedRequirements`: intersection of `registry.models[a].requirements || []` and `registry.models[b].requirements || []`
  - If `sharedSourceFiles.length > 0` OR `sharedPrefixes.length > 0`: include as candidate pair
- Return array sorted by `model_a` then `model_b`.

**3. `estimateMergedStateSpace(analysisA, analysisB)`**
Takes two model analysis objects (from `report.models[key]`).
- If both have non-null `estimated_states`: return `analysisA.estimated_states * analysisB.estimated_states`
- If either is `null`: return `null` (one or both unbounded/unresolvable)

**4. `classifyPair(pair, mergedStates, budget)`**
Takes a pair object, the merged state estimate (number or null), and the budget object `{ max_merged_states }`.
Returns an extended pair object with added fields:
- `estimated_merged_states`: the mergedStates value
- `recommendation`: `"merge"` if `mergedStates !== null && mergedStates <= budget.max_merged_states`, else `"interface-contract"`
- `rationale`: descriptive string explaining the recommendation:
  - merge: `"Combined state-space (${mergedStates}) is within 5-minute TLC budget (${budget.max_merged_states})"`
  - interface-contract with known states: `"Combined state-space (${mergedStates}) exceeds 5-minute TLC budget (${budget.max_merged_states})"`
  - interface-contract with null: `"One or both models have unbounded state-space; interface contract required"`

Also add a configurable constant near the existing `DEFAULT_THRESHOLDS`:

```javascript
  max_tlc_seconds: 300,
  throughput_states_per_sec: 10000,
  max_merged_states: 300 * 10000, // 3,000,000
};
```

**Add an `analyzeCrossModel(registry, models)` orchestrator function** (before `main()`):

```javascript
/**
 * Analyze cross-model relationships: detect pairs, estimate merged state-space,
 * and recommend merge or interface-contract.
 * @param {Object} registry — parsed model-registry.json
 * @param {Object} models — analyzed models (from analyzeModel)
 * @returns {{ budget: Object, pairs: Object[], summary: Object }}
 */
function analyzeCrossModel(registry, models) {
  const pairs = findCandidatePairs(registry, models);

  const classifiedPairs = pairs.map(pair => {
    const analysisA = models[pair.model_a];
    const analysisB = models[pair.model_b];
    const mergedStates = estimateMergedStateSpace(analysisA, analysisB);
    return classifyPair(pair, mergedStates, MERGE_BUDGET);
  });

  let mergeRecommended = 0;
  let interfaceContractNeeded = 0;
  for (const p of classifiedPairs) {
    if (p.recommendation === 'merge') mergeRecommended++;
    else interfaceContractNeeded++;
  }

  return {
    budget: { ...MERGE_BUDGET },
    pairs: classifiedPairs,
    summary: {
      total_pairs_analyzed: classifiedPairs.length,
      merge_recommended: mergeRecommended,
      interface_contract_needed: interfaceContractNeeded,
    },
  };
}
```

```javascript
describe('cross-model decomposition', () => {
  test('report has cross_model top-level key', () => {

    assert.strictEqual(budget.throughput_states_per_sec, 10000, 'Throughput should be 10,000 states/sec');

    const pairs = getReport().cross_model.pairs;

      assert.ok(pair.model_a, 'pair should have model_a');
      assert.ok(pair.model_b, 'pair should have model_b');
      assert.ok(pair.model_a < pair.model_b, 'model_a should sort before model_b: ' + pair.model_a + ' vs ' + pair.model_b);

      assert.ok(Array.isArray(pair.shared_requirement_prefixes), 'pair should have shared_requirement_prefixes array');
      assert.ok(Array.isArray(pair.shared_requirements), 'pair should have shared_requirements array');
      assert.ok(pair.shared_source_files.length > 0 || pair.shared_requirement_prefixes.length > 0,
        'pair should have shared sources or prefixes: ' + pair.model_a + ' + ' + pair.model_b);
      assert.ok(['merge', 'interface-contract'].includes(pair.recommendation),
        'recommendation should be merge or interface-contract, got: ' + pair.recommendation);
      assert.ok(typeof pair.rationale === 'string' && pair.rationale.length > 0,

    assert.strictEqual(total, cm.summary.total_pairs_analyzed,
```
