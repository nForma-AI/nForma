1. After the model file existence check and before the `extraArgs`/`prismArgs` build section, add a `readScoreboardRates()` helper function:

```js
// ── Read aggregate TP/unavail rates from quorum scoreboard ───────────────────
function readScoreboardRates(cwd) {
  const scoreboardPath = require('path').join(cwd, '.planning', 'quorum-scoreboard.json');
  // Non-Claude slots — Claude's self-vote is structural, not externally measured
  const SLOTS = ['gemini', 'opencode', 'copilot', 'codex'];
  const TP_PRIOR = 0.85;
  const UNAVAIL_PRIOR = 0.15;

  if (!require('fs').existsSync(scoreboardPath)) {
    process.stderr.write('[run-prism] No scoreboard found — using conservative priors\n');
    return { tpRate: TP_PRIOR, unavailRate: UNAVAIL_PRIOR, usedPrior: true };
  }

  let scoreboard;
  try {
    scoreboard = JSON.parse(require('fs').readFileSync(scoreboardPath, 'utf8'));
  } catch (_) {
    process.stderr.write('[run-prism] Failed to parse scoreboard — using conservative priors\n');
    return { tpRate: TP_PRIOR, unavailRate: UNAVAIL_PRIOR, usedPrior: true };
  }

  const rounds = scoreboard.rounds || [];
  if (rounds.length === 0) {
    process.stderr.write('[run-prism] Scoreboard has no rounds — using conservative priors\n');
    return { tpRate: TP_PRIOR, unavailRate: UNAVAIL_PRIOR, usedPrior: true };
  }

  // Compute per-slot rates then average across slots
  let tpSum = 0;
  let unavailSum = 0;
  let slotCount = 0;
  for (const slot of SLOTS) {
    const participated = rounds.filter(r => {
      const v = r.votes && r.votes[slot];
      return v !== undefined && v !== '' && v !== 'UNAVAILABLE';
    });
    const n = participated.length;
    if (n === 0) continue;
    const approveCount = participated.filter(r => r.votes[slot] === 'TP' || r.votes[slot] === 'TP+').length;
    const unavailCount = participated.filter(r => r.votes[slot] === 'UNAVAIL').length;
    tpSum += approveCount / n;
    unavailSum += unavailCount / n;
    slotCount++;
  }

  if (slotCount === 0) {
    process.stderr.write('[run-prism] No slot data in scoreboard — using conservative priors\n');
    return { tpRate: TP_PRIOR, unavailRate: UNAVAIL_PRIOR, usedPrior: true };
  }

  const tpRate     = parseFloat((tpSum / slotCount).toFixed(4));
  const unavailRate = parseFloat((unavailSum / slotCount).toFixed(4));
  return { tpRate, unavailRate, usedPrior: false };
}
```
