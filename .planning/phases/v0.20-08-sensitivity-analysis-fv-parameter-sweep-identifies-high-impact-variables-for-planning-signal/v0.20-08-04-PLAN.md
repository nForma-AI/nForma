    ## 8.3. FV Sensitivity Sweep (SENS-02 — fail-open)

    After the planner returns and before envelope update:

    ```bash
    # Run sensitivity sweep (fail-open — captures parameter brittleness before quorum)
    SENSITIVITY_REPORT_PATH="${PHASE_DIR}/sensitivity-report.ndjson"
    SENSITIVITY_MD_PATH="${PHASE_DIR}/sensitivity-report.md"
    SENSITIVITY_CONTEXT=""

    if [ -f "bin/run-sensitivity-sweep.cjs" ]; then
      node bin/run-sensitivity-sweep.cjs \
        2>/dev/null || true  # fail-open: sweep failure does not block workflow

      if [ -f "$SENSITIVITY_REPORT_PATH" ]; then
        # Extract top-3 high-sensitivity parameters for quorum context injection
        SENSITIVITY_CONTEXT=$(node -e "
          const fs = require('fs');
          const lines = fs.readFileSync('$SENSITIVITY_REPORT_PATH', 'utf8')
            .trim().split('\n').filter(l => l);
          const records = lines.map(l => { try { return JSON.parse(l); } catch { return null; } }).filter(Boolean);
          const flips = {};
          for (const r of records) {
            const p = r.metadata && r.metadata.parameter;
            if (!p) continue;
            if (!flips[p]) flips[p] = { count: 0, values: [] };
            if (r.metadata.delta && r.metadata.delta.startsWith('flip-to-')) {
              flips[p].count++;
              flips[p].values.push(r.metadata.value + '→' + r.result);
            }
          }
          const ranked = Object.entries(flips)
            .sort((a, b) => b[1].count - a[1].count)
            .slice(0, 3);
          if (ranked.length === 0) {
            console.log('SENSITIVITY_CONTEXT: no outcome flips detected (tools may not be installed)');
          } else {
            const lines = ['SENSITIVITY_CONTEXT (top-' + ranked.length + ' sensitive parameters):'];
            for (const [param, data] of ranked) {
              lines.push('  ' + param + ': ' + data.count + ' flip(s) at [' + data.values.join(', ') + ']');
            }
            console.log(lines.join('\n'));
          }
        " 2>/dev/null || echo "SENSITIVITY_CONTEXT: sweep data unavailable")

        # Also generate human-readable report (fail-open)
        node bin/sensitivity-report.cjs \
          2>/dev/null || true
      fi
    fi
    ```
