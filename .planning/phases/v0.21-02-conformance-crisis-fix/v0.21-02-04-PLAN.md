```javascript
// Group events by round_id for session-based validation
const rounds = new Map(); // round_id → events[]
for (const event of parsedEvents) {
  const key = event.round_id || event.session_id || '__standalone__' + event._lineIndex;
  if (!rounds.has(key)) rounds.set(key, []);
  rounds.get(key).push(event);
}

let mappedEventCount = 0; // denominator: events with valid XState mapping
let divergenceCount  = 0;

// Replay each round through a single actor — preserves cross-event context
for (const [roundId, roundEvents] of rounds) {
  const actor = createActor(qgsdWorkflowMachine);
  actor.start();
  for (const event of roundEvents) {
    const xstateEvt = mapToXStateEvent(event);
    if (!xstateEvt) continue; // skip unmappable (not counted in denominator)
    mappedEventCount++;
    const beforeSnap = actor.getSnapshot();
    actor.send(xstateEvt);
    const afterSnap  = actor.getSnapshot();
    // Divergence: state after send does not match expected
    // (implement expected-state check using existing mapToExpectedState or similar)
    if (/* state diverged */ false) {
      divergenceCount++;
      divergences.push(buildTTrace(event, afterSnap.value, expected, 'state_mismatch', scoreboardMeta, confidence, walker, qgsdWorkflowMachine, roundEvents));
    }
  }
  actor.stop();
}

// Rate = divergenceCount / mappedEventCount (NOT total events — unmappable excluded)
const divergenceRate = mappedEventCount > 0 ? divergenceCount / mappedEventCount : 0;
```
